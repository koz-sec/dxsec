        # ツイートの分析

        前回、キーワード検索を実施した。

        import tweepy

# 検索ワードを決めて、
searchword = 'ランサムウェア -filter:replies'

# 認証情報を入力して、
api_key = '私のAPIキー'
api_secret = '私のAPIキー'
access_token = '私のトークンキー'
access_token_secret = '私のトークンキー'

# Twitterで認証して
auth = tweepy.OAuthHandler(api_key, api_secret)
auth.set_access_token(access_token, access_token_secret)
api = tweepy.API(auth, wait_on_rate_limit = True)
    
# ツイートを取得する。
tweets = api.search_tweets(q=searchword, result_type='recent', count=20)
    
# その結果は？
print(tweets)

        個人情報の扱い

        IDやURLといった情報で個人の特定に繋がり得る可能性があるため、「XXXXX」に書き換えている。


        JSONオブジェクトがわかる人なら、じっくり眺めると欲しいデータがどこにあるか分かる。

        例えば、ツイートの本文を取り出したければ、 tweets[0]._json['text'] とすれば良いし、URLを取り出したいなら tweets[0]._json['entities']['urls'][0]['url'])と書けば良い。

        上記プログラムの変数「tweets」には、ツイート1件ごとに配列で保管されている。 以下のようなfor文を作れば、繰り返し処理ができるはず。(このままだとエラーが出る。エラー処理済みのプログラムは後述)

        for a_tweet in tweets:
    URL = str(a_tweet._json['entities']['urls'][0]['expanded_url'])
    print(URL)

        # URLだけを抽出

        情報漏えい のキーワードで検索すると、膨大な量のツイートが見つかる。 取り出したツイートから、URLだけを取り出す。
        URLがTwitterから始まるものは、他のツイートを引用しているだけなので、私が求めている 情報漏えいに関する内容が書かれたウェブサイトのURL ではない。除外する。

        以下のプログラムでURLだけを抜き出す。

        q = "情報漏えい exclude:retweets"
tweets = api.search_tweets(q=q, tweet_mode='extended', result_type="recent", include_entities=True, count=10)

cnt = 0
# 表示する。
for a_tweet in tweets:
    print('---'+str(cnt+1)+'件目---')
    if 0 != len(a_tweet._json['entities']['urls']): # URLが存在すれば
        URL = str(a_tweet._json['entities']['urls'][0]['expanded_url'])
        if not URL.startswith('https://twitter.com'): # URLがTwitterから始まらなければ
            print(URL)
    cnt = cnt + 1
